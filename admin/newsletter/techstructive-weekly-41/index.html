  • Removing duplicates from a list of dictionaries 
    def remove_duplicate_dicts(dict_list):
    seen = set()
    unique_dicts = []
    for d in dict_list:
        dict_tuple = tuple(sorted(d.items()))
        if dict_tuple not in seen:
            seen.add(dict_tuple)
            unique_dicts.append(d)
    return unique_dicts
